# Skeleton

Use to show a placeholder while content is loading.

## Installation

```bash
npx shadcn-lynx add skeleton
```

## Usage

```typescript
import { Skeleton } 

export function App() {
  return (
    <view>
      <Skeleton className="h-5 w-[200px]" />
    </view>
  )
}
```

import { ComponentPreview } from '@/components/component-preview'
import { SkeletonDefaultDemo, SkeletonCardDemo, SkeletonTextDemo } from '@/examples/skeleton-demo'
