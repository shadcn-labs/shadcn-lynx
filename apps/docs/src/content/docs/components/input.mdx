# Input

Displays a form input field or a component that looks like an input field.

## Installation

```bash
npx shadcn-lynx add input
```

## Usage

```typescript
import { Input } from '@/components/ui/input'

export function App() {
  return (
    <view>
      <Input placeholder="Enter text..." />
    </view>
  )
}
```

import { ComponentPreview } from '@/components/component-preview'
import { InputPreview } from '@/components/previews/input-preview'
import { LabelPreview } from '@/components/previews/label-preview'

## Examples

### Default

<ComponentPreview
  component={() => <InputPreview placeholder="Enter your email" />}

<ComponentPreview

## API Reference

### Props

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| type | `text` \| `password` \| `email` \| `number` \| `tel` | `text` | Input type |
| placeholder | `string` | - | Placeholder text |
| disabled | `boolean` | `false` | Whether the input is disabled |
| value | `string` | - | Controlled value |
| defaultValue | `string` | - | Default value |
| onChange | `(value: string) => void` | - | Change handler |
| className | `string` | - | Additional CSS classes |
| ...props | `InputProps` | - | Native input props from @lynx-js/lynx-ui |
